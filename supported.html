Supported queries

Youâ€™ll find below some examples of the kind of supported queries:

scala> import com.tecsisa.lightql.parser._
import com.tecsisa.lightql.parser._

scala> parse("foo = 25") // filtered (a.k.a. exact query)
res0: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,25)),8)

scala> parse("foo != 25") // not equals
res1: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,!=,25)),9)

scala> parse("foo ~ 25") // match query
res2: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,~,25)),8)

scala> parse("foo !~ 25") // not matches
res3: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,!~,25)),9)

scala> parse("foo.name = \"foobar\"") // nested
res4: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo.name,=,foobar)),19)

scala> parse("foo > 25") // range (greater than)
res5: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,>,25)),8)

scala> parse("foo < 25") // range (less than)
res6: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,<,25)),8)

scala> parse("foo >= 25")
res7: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,>=,25)),9)

scala> parse("foo <= 25")
res8: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,<=,25)),9)

scala> parse("foo = 25 and foo = 10") // combined
res9: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(CombinedClause(Clause(foo,=,25),and,Clause(foo,=,10))),21)

scala> parse("foo = 25 or foo = 10")
res10: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(CombinedClause(Clause(foo,=,25),or,Clause(foo,=,10))),20)

scala> parse("foo = 25 or foo = 10 and bar = 40")
res11: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(CombinedClause(CombinedClause(Clause(foo,=,25),or,Clause(foo,=,10)),and,Clause(bar,=,40))),33)

scala> parse("foo = 25 or (foo = 10 and bar = 40)") // as in SQL, the grammar is left recursive
res12: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(CombinedClause(Clause(foo,=,25),or,CombinedClause(Clause(foo,=,10),and,Clause(bar,=,40)))),35)
